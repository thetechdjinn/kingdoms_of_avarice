Energy Formulas

effectiveEnergy = baseEnergy * combatMult * levelMult * dexMult * encMult

levelMult = 1 + (charLevel - 1) * 0.02

dexMult = 1 + max(0, (dex - 50) / 10) * 0.01

encMult = max(0.5, 1 + (0.5 - encRatio) * 0.5)

rawSwings = effectiveEnergy / weaponSpeed

swings = min(rawSwings, maxAttacks)

Critical Hit Formulas (MajorMUD-style)

statCrit = floor(level/10) + max(0, floor((INT-50)/10)) + max(0, floor((DEX-50)/25))

encCrit = 20 if enc ≤ 32%, 10 if enc ≤ 65%, else 0

preCap = statCrit + encCrit + classBonus + weaponCrit

finalCrit = if (preCap > 40) then 40 + floor((preCap - 40) / 3) else preCap

finalCrit = max(0, min(60, finalCrit))

critDamage = maxWeaponDamage * random(2.0 to 4.0)

Dodge Formulas (MajorMUD-style)

baseDodge = classDodgeBonus + raceDodgeBonus + equipDodgeBonus

statDodge = floor(DEX/10) * 2 + floor(CHA/10) * 1

preCap = baseDodge + statDodge

postCap = if (preCap > 52) then 52 + floor((preCap - 52) / 4) else preCap

effectiveDodge = if (baseDodge = 0 or attackerAcc ≤ 8) then 0 else max(0, min(90, floor(postCap * 10 / attackerAcc)))
